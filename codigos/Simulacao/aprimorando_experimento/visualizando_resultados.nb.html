# Preparar base formatada
tabela_formatada <- metricas_resultado %>%
  filter(estatistica == "imputado_media") %>%
  mutate(
    metodo = factor(metodo, levels = ordem_metodos),
    grupo_coluna = paste(mecanismo, proporcao, sep = "_"),
    Erro_Bruto = format(round(`Erro Bruto`, 3), nsmall = 3, decimal.mark = ","),
    Desvio_Padrao = format(round(`Desvio Padrão`, 3), nsmall = 3, decimal.mark = ",")
  ) %>%
  select(metodo, grupo_coluna, Erro_Bruto, Desvio_Padrao) %>%
  arrange(metodo)

# Tabela de Erro Bruto
