🔹 Choosing the Best Model
Linear Regression: Works well if relationships are linear.
Random Forest Regressor: Better for nonlinear relationships.
k-NN Imputer: If no clear relationships exist.


from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

# Select feature columns and target column (A, which has missing values)
df_train = df[df['A'].notna()]  # Rows where A is NOT missing
df_missing = df[df['A'].isna()]  # Rows where A is missing

# Define X (features) and y (target)
X_train = df_train[['B', 'C']]  # Use other columns as features
y_train = df_train['A']  # Target column

# Train Linear Regression model
model = LinearRegression()
model.fit(X_train, y_train)

# Predict missing values
X_missing = df_missing[['B', 'C']]
predicted_values = model.predict(X_missing)

# Fill missing values in A with predicted values
df.loc[df['A'].isna(), 'A'] = predicted_values

print(df)
